Detectar cron


SCRIPT BASH PARA DETECTAR CRON

#!/bin/bash

old_process=$(ps -eo command)

while true; do
  new_process=$(ps -eo command)
  diff <(echo “$old_process”) <(echo “$new_process”) | grep “[\>\<]”
  old_process=$new_process
done
